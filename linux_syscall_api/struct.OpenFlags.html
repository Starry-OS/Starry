impl OpenFlags

pub const RDONLY: OpenFlags = _

只读

pub const WRONLY: OpenFlags = _

只能写入

pub const RDWR: OpenFlags = _

读写

pub const CREATE: OpenFlags = _

如文件不存在，可创建它

pub const EXCLUSIVE: OpenFlags = _

确认一定是创建文件。如文件已存在，返回 EEXIST。

pub const NOCTTY: OpenFlags = _

使打开的文件不会成为该进程的控制终端。目前没有终端设置，不处理

pub const TRUNC: OpenFlags = _

同上，在不同的库中可能会用到这个或者上一个

pub const NON_BLOCK: OpenFlags = _

非阻塞读写?(虽然不知道为什么但 date.lua 也要) 在 socket 中使用得较多

pub const TEXT: OpenFlags = _

要求把 CR-LF 都换成 LF

pub const BINARY: OpenFlags = _

和上面不同，要求输入输出都不进行这个翻译

pub const DSYNC: OpenFlags = _

对这个文件的输出需符合 IO 同步一致性。可以理解为随时 fsync

pub const NOFOLLOW: OpenFlags = _

如果是符号链接，不跟随符号链接去寻找文件，而是针对连接本身

pub const CLOEXEC: OpenFlags = _

在 exec 时需关闭

pub const DIR: OpenFlags = _

是否是目录

impl OpenFlags

pub const fn iter(&self) -> Iter<OpenFlags>

Yield a set of contained flags values.

Each yielded flags value will correspond to a defined named flag. Any unknown bits will be yielded together as a final flags value.

pub const fn iter_names(&self) -> IterNames<OpenFlags>

Yield a set of contained named flags values.

This method is like iter, except only yields bits in contained named flags. Any unknown bits, or bits not corresponding to a contained flag will not be yielded.
